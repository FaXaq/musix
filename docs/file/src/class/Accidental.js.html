    const accidentals = ['s', 'ss', 'b', 'bb', 'n', '♯', '𝄪', '♭', '𝄫', '♮'];

module.exports = class Accidental {

  static validate(accidental) {
    if (accidentals.includes(accidental)) return accidental
    return new Error('No accidental with character', accidental)
  }

}
